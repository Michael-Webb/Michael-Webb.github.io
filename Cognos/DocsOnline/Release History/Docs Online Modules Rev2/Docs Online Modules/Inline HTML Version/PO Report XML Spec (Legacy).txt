  console.log(&quot;NodeList: &quot;, nL);

  nL.forEach((node) =&gt; {

      //console.log(&apos;Beginning Catch&apos;)
    }
    try {
      let check_img = document.getElementById(
        `${this.urldec(node.dataset.unid)}` + &quot;-link&quot;
      );
      check_href = check_img.getAttribute(&quot;href&quot;) === &quot;&quot;;
      //console.log(check_href);
      return;
    } catch {
      // If no img is in the span then add the clock image to the element with a unique ID and add a hyperlink with src=&quot;&quot; within it
      let add_timer = document.createElement(&quot;img&quot;);
      //If you choose to change the icon for loading to a different dimension this is where you can change the length and width through multiplying the &apos;node.dataset.font&apos; by the value you want one dimension changed by
      add_timer.width = node.dataset.font;
      add_timer.height = node.dataset.font;
      add_timer.src = `${this.urldec(node.dataset.clock)}`;
      add_timer.id = `${this.urldec(node.dataset.unid)}`;
      //console.log(&apos;ids&apos;, `${node.dataset.unid}`)
      let add_link = document.createElement(&quot;a&quot;);
      add_link.setAttribute(&quot;href&quot;, &quot;&quot;);
      add_link.setAttribute(&quot;target&quot;, &quot;_blank&quot;);
      add_link.setAttribute(
        &quot;id&quot;,
        `${this.urldec(node.dataset.unid)}` + &quot;-link&quot;
      );
      node.appendChild(add_link);
      add_link.appendChild(add_timer);

      //Execute a fetch request per node iterated on. Fetch(ServerUrl + arguments + environment) headers: token, user
      fetch(
        `${
          this.urldec(node.dataset.server) +
          encodeURI(
            this.fEncode(this.urldec(node.dataset.arg).replaceAll(&quot;\\&quot;, &quot;&quot;))
          ) +
          `${this.urldec(node.dataset.env)}`
        }`,

          method: &quot;GET&quot;,
          headers: {
            token: `${node.dataset.token}`,
            user: `${this.urldec(node.dataset.user)}`,
          },
        }
      )
        .then((response) =&gt; {
          let res = response.text();
          //console.log(res)
          return res;
